And initialize plugin

            
$(document).ready(function() {
  $('.panel-group').accordionWithLinks({
    duration: 400
  });
});

All supported params:

            
$('.panel-group').accordionWithLinks({
  easing: 'swing'
});

In this option you can change amount of show

            
$('.panel-group').accordionWithLinks({
  divHeading: '.panel-heading'
});
            
          
            
$('.panel-group').accordionWithLinks({
  divCollapse: '.collapse'
});
